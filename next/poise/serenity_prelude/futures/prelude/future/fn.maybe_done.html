Wraps a future into a MaybeDone

§Examples

use futures::future;
use futures::pin_mut;

let future = future::maybe_done(async { 5 });
pin_mut!(future);
assert_eq!(future.as_mut().take_output(), None);
let () = future.as_mut().await;
assert_eq!(future.as_mut().take_output(), Some(5));
assert_eq!(future.as_mut().take_output(), None);
