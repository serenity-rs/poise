    commands: &[Command<U, E>],

Expand description

Collects all commands into a Vec<serenity::CreateCommand> builder, which can be used to register the commands on Discord

Also see register_application_commands_buttons for a ready to use register command

let commands = &ctx.framework().options().commands;
let create_commands = poise::builtins::create_application_commands(commands);

serenity::Command::set_global_commands(ctx, create_commands).await?;
